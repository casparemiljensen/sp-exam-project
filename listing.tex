  D_A     >> (50.00)  >>= DA + A

  D_R     >> (100.00) >>= DR + A
  D_A     >> (500.00) >>= MA + D_A
  DA      >> (50.00)  >>= MA + DA
  D_R     >> (50.00)  >>= MR + D_R
  DR      >>  (0.01)  >>= MR + DR
  MA      >> (50.00)  >>= MA + A
  MR      >>  (5.00)  >>= MR + R
  A + R   >>  (2.00)  >>= C
  C       >>  (1.00)  >>= R
  A       >>  (1.00)  >>= env
  R       >>  (0.20)  >>= env
  MA      >> (10.00)  >>= env
